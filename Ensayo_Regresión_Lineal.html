import numpy as np
import matplotlib.pyplot as plt
import pandas as pd

def mco(X,y):
  X = np.c_[np.ones(len(X)), X]
  return np.dot(np.linalg.inv(np.dot(X.T,X)),np.dot(X.T,y))
